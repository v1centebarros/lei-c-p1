NUM position= 0;

NUM bn = 1;
PRINT "..Rotating to beacon\n";

ROTATE 80 UNTIL (-10 < beaconAngle(bn) < 10); #ja faz o apply()
# ha apply() com argumento, é importante ou podemos fazer o apply automatico com um valor default?

PRINT "\n..Going to beacon\n";

WHILE groundType() != bn DO
    NUM linearPower = 80.0;
    NUM angularPower = 1 * beaconAngle(bn);
    MOVE (linearPower - angularPower, linearPower + angularPower); # or MOVE AT ja faz o apply()
END;

STOP; #MOVE(0,0)

pickUp();

returning();

ROTATE 80 UNTIL (-10 < startAngle() <10);

WHILE startDistance() > 2 DO
    NUM linearPower = 80.0;
    NUM angularPower = 1 * startAngle();
    MOVE (linearPower - angularPower, linearPower + angularPower);
END;

STOP;

finish();
